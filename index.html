local Assets = require(script.Parent.Assets) workspace.StreamingEnabled = true local Lighting = game:GetService("Lighting") Lighting.ClockTime = 14 Lighting.Brightness = 2 Lighting.FogEnd = 900 Lighting.FogColor = Color3.fromRGB(180, 220, 255) local atmosphere = Instance.new("Atmosphere") atmosphere.Density = 0.35 atmosphere.Offset = 0.25 atmosphere.Color = Color3.fromRGB(199, 220, 255) atmosphere.Decay = Color3.fromRGB(90, 130, 190) atmosphere.Glare = 0 atmosphere.Haze = 1.5 atmosphere.Parent = Lighting local WORLD_SIZE = 600 local HALF = WORLD_SIZE / 2 workspace.Terrain:FillBlock( CFrame.new(0, -4, 0), Vector3.new(WORLD_SIZE, 8, WORLD_SIZE), Enum.Material.Grass ) local worldFolder = Instance.new("Folder") worldFolder.Name = "WorldObjects" worldFolder.Parent = workspace local objectCounts = { Trees = 120, Bushes = 250, Rocks = 90, Grass = 500, } local function randomPos() return Vector3.new( math.random(-HALF, HALF), 0, math.random(-HALF, HALF) ) end local function spawnTree(pos) local data = Assets.Trees[math.random(#Assets.Trees)] local model = Instance.new("Model") model.Name = "Tree" local trunk = Instance.new("Part") trunk.Size = data.trunk.size trunk.Color = data.trunk.color trunk.Material = data.trunk.material trunk.Anchored = true trunk.CastShadow = false trunk.Position = pos + Vector3.new(0, data.trunk.size.Y / 2, 0) trunk.Parent = model local leaves = Instance.new("Part") leaves.Shape = Enum.PartType.Ball leaves.Size = data.leaves.size leaves.Color = data.leaves.color leaves.Material = data.leaves.material leaves.Anchored = true leaves.CastShadow = false leaves.Position = pos + Vector3.new(0, data.trunk.size.Y + data.leaves.size.Y * 0.4, 0) leaves.Parent = model model.Parent = worldFolder end local function spawnBush(pos) local data = Assets.Bushes[math.random(#Assets.Bushes)] local part = Instance.new("Part") part.Shape = Enum.PartType.Ball part.Size = data.size part.Color = data.color part.Material = data.material part.Anchored = true part.CastShadow = false part.Position = pos + Vector3.new(0, data.size.Y * 0.4, 0) part.Name = "Bush" part.Parent = worldFolder end local function spawnRock(pos) local data = Assets.Rocks[math.random(#Assets.Rocks)] local part = Instance.new("Part") part.Size = data.size part.Color = data.color part.Material = data.material part.Anchored = true part.CastShadow = false part.Position = pos + Vector3.new(0, data.size.Y / 2, 0) part.Rotation = Vector3.new(math.random(-15, 15), math.random(0, 360), math.random(-15, 15)) part.Name = "Rock" part.Parent = worldFolder end local function spawnGrass(pos) local data = Assets.Grass[math.random(#Assets.Grass)] local part = Instance.new("Part") part.Size = data.size part.Color = data.color part.Material = data.material part.Anchored = true part.CastShadow = false part.Position = pos + Vector3.new(0, data.size.Y / 2, 0) part.Name = "Grass" part.Parent = worldFolder end local spawners = { Trees = spawnTree, Bushes = spawnBush, Rocks = spawnRock, Grass = spawnGrass, } for category, count in pairs(objectCounts) do for i = 1, count do spawners[category](randomPos()) if i % 15 == 0 then task.wait() end end end
